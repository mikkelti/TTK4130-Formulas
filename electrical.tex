\part{Electrical and mechanical}

\section{Capacitor}
\begin{align}
    i(t) &= C \frac{dv_C(t)}{dt}
\end{align}

\section{Inductor}
\begin{align}
    v(t) &= L \frac{di_L(t)}{dt}
\end{align}

\section{Rotating object}
\begin{align}
    \omega = \frac{v}{r}
\end{align}
where $r$ is the distance from axis, $v$ is the tangential speed and $\omega$ is the angular frequency of rotation.
